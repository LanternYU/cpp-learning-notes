    private:
    std::vector<T> data;
    public:
    ...

### 4.1.2 Composition下的构造与析构
构造由内而外：Container的构造函数首先调用Component的default构造函数，然后再执行自己//由内而外搭建

析构由外而内：Container的构造函数首先调用自己的析构函数，然后再执行component的析构//由外而内拆解

## 4.3 继承(Inheritance is-a)
```cpp
class Shape {
   protected:
      int width, height;
   public:
      Shape(int a = 0, int b = 0) : width(a), height(b) { }
      virtual int area() {
         cout << "Shape class area: " << endl;
         return 0;
      }
};
 

class Rectangle : public Shape {
   public:
      Rectangle(int a = 0, int b = 0) : Shape(a, b) { }
      int area() override { 
         cout << "Rectangle class area: " << endl;
         return width * height;
      }
};
 
class Triangle : public Shape {
   public:
      Triangle(int a = 0, int b = 0) : Shape(a, b) { }
      int area() override { 
         cout << "Triangle class area: " << endl;
         return (width * height / 2); 
      }
};
 

int main() {
   Shape *shape;           // 基类指针
   Rectangle rec(10, 7);   // 矩形对象
   Triangle tri(10, 5);    // 三角形对象
 
   shape = &rec;
   cout << "Rectangle Area: " << shape->area() << endl;
   shape = &tri;
   cout << "Triangle Area: " << shape->area() << endl;
 
   return 0;
}
```
### 4.3.2 继承下的构造和析构
```CPP
//构造由内而外
Derived::Derived(...){}
Base::Base(){}
//析构由外而内
Derived::~Derived(...){~Base()}
//Base类的析构函数要设置成virtual
```
### 4.3.3 虚函数、虚指针、虚函数表

虚函数分为三类  
non-vitrual函数：不希望子类override  
virtual函数：希望子类override，且本身已经有定义  
pure virtual函数：希望子类一定要override，且它没有定义
```cpp
class Shape{
    public:
    virtual void draw() = 0 ;//pure virtual
    virtual void error(const std::string& msg);//virtual
    int objectID() const;//non-vitrual
    ...
}
```
